\crule{Testsignale}

\begin{itemize}
	\item Kronecker Delta

	      \begin{minipage}{0.4\textwidth}\vspace*{0pt}
		      $$
			      \delta[n] = \begin{cases}
				      1 \text{ wenn } n=0 \\ 0 \text{ sonst}
			      \end{cases}
		      $$
	      \end{minipage}
	      \begin{minipage}{0.5\textwidth}\vspace*{0pt}
		      \centering
		      \begin{tikzpicture}[scale=0.7]
			      \node at (0, 2) {$\delta [n]$};
			      \ax{$n$}\nakedsequence{{0/-3, 0/-2, 0/-1, 1/0, 0/1, 0/2, 0/3}};
		      \end{tikzpicture}
	      \end{minipage}

	\item Sprungfunktion

	      \begin{minipage}{0.4\textwidth}\vspace*{0pt}
		      $$
			      u[n] = \begin{cases}
				      1 \text{ wenn } n\geq 0 \\ 0 \text{ sonst}
			      \end{cases}
		      $$
	      \end{minipage}
	      \begin{minipage}{0.5\textwidth}\vspace*{0pt}
		      \centering
		      \begin{tikzpicture}[scale=0.7]
			      \node at (0, 2) {$u [n]$};
			      \ax{$n$}\nakedsequence{{0/-3, 0/-2, 0/-1, 1/0, 1/1, 1/2, 1/3}};
		      \end{tikzpicture}
	      \end{minipage}

	\item Dirac-Distribution

	      \begin{minipage}{0.4\textwidth}\vspace*{0pt}
		      $$
			      \delta(t) = \begin{cases}
				      \to \infty & \text{wenn } t=0 \\ 0 & \text{sonst}
			      \end{cases}
		      $$
	      \end{minipage}
	      \begin{minipage}{0.5\textwidth}\vspace*{0pt}
		      \centering
		      \begin{tikzpicture}[scale=0.7]
			      \node at (0, 2) {$\delta (t)$};
			      \ax{$t$}\draw[->, blue, very thick] (0, 0) -- (0,1);
		      \end{tikzpicture}
	      \end{minipage}

	      Abtasteigenschaft der Dirac-Distribution:

	      \begin{equation}
		      f(t_{0}) = \int_{-\infty}^{\infty}f(t)\cdot\delta(t-t_{0})dt
		      \label{eq:abtasteigenschaft}
	      \end{equation}

\end{itemize}

\crule{Geometrische Reihe}

\begin{itemize}
	\item  Unendlicher Grenzwert

	      \begin{equation}
		      \sum_{i=0}^{\infty}q^i = \frac{1}{1-q}
		      \label{eq:geom_infty}
	      \end{equation}

	\item Partialsumme der geometrischen Reihe

	      \begin{equation}
		      \sum_{i=0}^k q^i = \frac{1-q^{k+1}}{1-q}
		      \label{eq:geom_n}
	      \end{equation}

\end{itemize}

\crule{Trigonometrie}

\begin{itemize}
	\item Komplexer Zusammenhang des Kosinus
	      \begin{equation}
		      \cos(x) = \frac{e^{jx}+e^{-jx}}{2}
		      \label{eq:cplxcos}
	      \end{equation}

	\item Eulersche Formel
	      \begin{equation}
		      e^{jx} = \cos(x) + j\sin(x)
		      \label{eq:euler}
	      \end{equation}

	\item Eulersche Identität
	      \begin{equation}
		      e^{-j\pi} = -1
		      \label{eq:euler-ident}
	      \end{equation}
\end{itemize}
